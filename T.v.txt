module T_flipflop (
    input clk,
    input rst_n, 
    input t,
    output reg q,
    output q_bar
);

always @(posedge clk or negedge rst_n) begin
  if (!rst_n) begin 
        q <= 1'b0;
    end else begin
        if (t) begin // T=1, toggle
            q <= ~q;
        end else begin // T=0, maintain state
            q <= q;
        end
    end
end

assign q_bar = ~q; // Output complement of q

endmodule
